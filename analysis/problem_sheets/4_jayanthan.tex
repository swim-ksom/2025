\begin{questions}
	\question Give an example of an unbounded sequence that has a convergent subsequence.
	\question Let $\{x_n\}$ be a sequence of real numbers. Which of the following are true?
	\begin{parts}
		\part All subsequnces of $x_n$ converges implies $x_n$ also converges.
		\part All convergent subsequences of $x_n$ converges to $x$ implies $x_n$ converges to $x$.
		\part Any subsequence of $x_n$ has a subsequence converges to $x$, then $x_n$ converges to $x$.
		\part $\{x_n\}$ is bounded and all of its convergent subsequences converges to $x$, implies $x_n$ conveges to $x$.
	\end{parts}
	\question Construct a sequence $\{x_n\}$ such that for any natural number $N$, there exits a subsequence of $\{x_n\}$ converging to $N$.
